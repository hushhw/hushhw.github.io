## AC代码

```c++
#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
using namespace std;
 
long long a[10005];  

int main()  
{  
	int n;  
	while(scanf("%d",&n)!=EOF)  
	{  
		a[0]=0;  
		a[1]=3;  
		a[2]=6;  
		a[3]=6;  
		for(int i=4;i<=n;i++)  
		{  
			a[i]=a[i-1]+2*a[i-2];  
		}  
		cout<<a[n]<<endl;  
	}  
	return 0;  
}  
```
